/**
 * @module studentRegStart
 */
import SelectUSState from 'react-select-us-states'
import PhoneInput from 'react-phone-number-input/input'

/**
 * This function provides the HTML formatting for the first page of the student registration process
 * @param {State} props - State information from register.js that calls this function
 * @returns {HTMLCollection}
 */
function StudentRegStart(props) {
    return (
        <>
            <label>
                User Name:
                <input 
                    value={props.state.username}
                    id="username"
                    type="text"
                    minLength="5"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>
            <label>
                First Name:
                <input value={props.state.firstName}
                    id="firstName"
                    type="text"
                    maxLength="30"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                Last Name:
                <input value={props.state.lastName}
                    id="lastName"
                    type="text"
                    maxLength="30"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                Email:
                <input value={props.state.email}
                    id="email"
                    type="email"
                    pattern="[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{1,63}$"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                Address:
                <input value={props.state.addressStreet}
                    id="addressStreet"
                    type="text"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                Address Line 2:
                <input value={props.state.addressStreet2}
                    id="addressStreet2"
                    type="text"
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                City:
                <input value={props.state.addressCity}
                    id="addressCity"
                    type="text"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                State:
                <SelectUSState value={props.state.addressState}
                    id="addressState"
                    type="text"
                    required
                    onLoad={e => props.handleStateStart(e)}
                    onChange={e => props.handleStateChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                Zip Code:
                <input value={props.state.addressZip}
                    id="addressZip"
                    type="text"
                    minLength="5"
                    maxLength="10"
                    pattern="[0-9-]*"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>

            <label>
                Phone Number:
                <input value={props.state.phoneNumber}
                    id="phoneNumber"
                    type="text"
                    minLength="10"
                    maxLength="10"
                    pattern="[0-9]*"
                    required
                    onChange={e => props.handleChange(e)}
                    onBlur={e => props.handleFocusChange(e)}
                />
            </label>
        </>
    )
}

export default StudentRegStart
